原生JScode簡介

此篇會逐一列出各種用途的code並簡易的附上說明。

指定DOM

    document.querySelectorAll('.class / #id') //取得所有元素並陣列化
    document.querySelector('.class / #id') //取得一個元素，若有多個只會抓第一個
    document.getElementById('id') //指定Id
    document.getElementsByTagName('標籤名稱')//對應標籤 如h1等
    document.getElementsByClassName('class名稱')//對應class名稱

陣列相關

    Array.push(變數 / 數字 / '字串'/ {陣列})
    Array.splice(第幾筆開始,刪除幾筆)
    Array.pop()//刪除陣列最後一筆並回傳陣列最後一筆
    Array.join('指定字串')//將陣列中元素、物件以('指定字串連接')並且轉為字串
    Array.length//陣列數量總數

物件相關

    Object.freeze//凍結物件屬性及值(通常搭配const)
    console.log(Object.isFrozen(物件))//偵測是否有被凍結,回傳boolean
    Object.defindproperty(指定物件,'指定物件的屬性',{
        value:'值',
        writable:boolean,//可否被寫入
        configurable:boolean,//可否被刪除
        enumerable:boolean,//可否被列舉
    });//物件的特徵修改
    Object.preventExtensions(物件)//防止擴充(新增)屬性,直接針對物件做限制
    console.log(Object.isExtensions(物件))//偵測是否可以被擴充,回傳boolean
    console.log(Object.getOwnpropertyDescriptor(物件,'屬性'))//偵測物件特徵,回傳boolean

DOM的設定

    Variable.getAttribute('屬性')//取屬性值 例如a裡面的href及class、id、data-type等
    Variable.setAttribute('屬性','屬性值')//賦予屬性值
    Variable.classlist.add('class')//新增個class於DOM上
    variable.classlist.remove('class')//移除指定DOM上的class
    document.classList.toggle('class名稱')//當DOM有這個class的話移除若沒有則新增 大概可以說是add及remove簡化版
    Variable.textContent = '字串'//改變內容文字
    variable.write(' 這裡寫你要呈現的文字 ')//新增文字，類innerHTML，可賦予標籤
    Variable.innerHTNL =//清空原先內容並加入文字or標籤
    Variable.appendChild(Variable)//上傳子節點(於最後)
    會與let Variable = document.createElement('標籤')//新增標籤配合
    先新增標籤變數，再上傳。
    Variable.insertBefore('標籤',位置)//上傳子節點並指定位置

event相關

    e.target.dataset.自訂ID //取自訂義data的值
    e.target.nodeName 取元素(標籤)名。都會是大寫
    e.target.className 取class名
    e.target.textContent 取點擊取得文字內容
    e.target.value 取值(常用於input)
    e.target.files//擷取input file的檔案
    e.dataTransfer.files//擷取拖曳的檔案
    e.stopPropagation() 阻止冒泡事件。點擊區塊內只執行一種event
    e.preventDefault() 取消元素默認事件。例如a連結、右鍵設定選單
    監聽事件種類:click點擊, change表單更動, input表單更動(與change相比，只要一更動就觸發),
    keydown按下鍵盤、keyup放開鍵盤, drop拖動放下、dragover拖動時,
    blur離開焦點, focus所在焦點, mousemove鼠標滑入觸發(類CSS的hover)
    DOM.addEventListener('事件',function,false/ture)//監聽事件設定

BOM

    window.history.forward()//到有紀錄的下一頁
    window.history.back()//回到有紀錄的上頁
    window.print()//列印
    window.href()//前往網址
    window.open('網址')//開新分頁並前往網址頁面(空值的話等於開新分頁)
    window.innerHeight//網頁高度
    window.innerWidth//網頁寬度
    window.outerHeight//整個瀏覽器高度
    window.outerWidth//整個瀏覽器寬度

獲取節點

    DOM.childNodes//得到DOM的全部子節點
    DOM.parentNode//得到DOM的父節點
    DOM.nextSbiling//獲得DOM的下一個兄弟節點
    DOM.previousSbiling//得到DOM的上一個兄弟節點
    DOM.firstChild//獲得DOM的第一個子節點
    DOM.lastChild;//獲得DOM的最後一個子節點

AJAX

    let xhr = new XMLHttpRequest();//建立連接請求
    xhr.open('get/post','網址',ture/false)//連線對象及作用
    xhr.setRequestHeader('Content-type','application/x-www-form-urlencoded')//設定傳送格式
    // 發送表單類型的資料：('Content-Type','application/x-www-form-urlencoded')
    // 發送 JSON 類型的資料：('Content-Type','application/json')
    xhr.send(內容)//上傳內容
    xhr.onload =function(){}//非同步時讀取資料用

未分類code

    parseInt(欲轉換內容)字串轉數字，不取小數點

    isNaN(value)//如果是NaN就回傳ture。空值亦可使用，可用在沒輸入任何東西的情況下
    localStorage.getItem("localName")//取出本地端localstorage
    localStorage.setItem("自定義localName",輸入的字串)//設定or加入localstorage
    let today = new Date()//取得目前時間 得先取得目前時間才可以用today.getHours等去取得細項
    getDay()取得星期, getFullYear()取得西元年, getMonth()取得月份,
    getDate()取得天數, getHours()取得小時

    scrollWidth:獲取物件的滾動寬度
